Stage 4 — Score Aggregation（融合得分聚合）¶

本阶段的作用是 将 pair-level 得分映射回原始 3D 检测框, 得到最终的每个 3D 框的融合分数。

本阶段包含两个算子：

  1. scatter_to_matrix
    将 p 个 pair 得分映射回 (K×N) score matrix。

  2. score_aggregator
    沿 2D 维度 MaxPool 得到最终 N 个融合得分。

@torch.library.custom_op("clocs::scatter_to_matrix", mutates_args=[])
def scatter_to_matrix(pair_scores: torch.Tensor,
                      pair_indices: torch.Tensor,
                      K: int,
                      N: int) -> torch.Tensor:
    """
    将 p 个 pair 得分根据 pair_indices 散射回 (1, K, N) 的得分矩阵。
    对于没有对应 pair 的位置填充一个极小值。

    Args:
        pair_scores: (p,) 或 (1,1,p)
            每个 pair 的融合得分。
        pair_indices: (p,2)
            每行 (i_2d, j_3d) 表示该 score 应写入矩阵位置 (i, j)。
        K: int
            本帧 2D 框数量上限。
        N: int
            本帧 3D 框数量上限。

    Returns:
        score_matrix: (1, K, N)
            对应位置写入 pair score，其余位置为负无穷小。
    """

    # 兼容输入为 (1,1,p) 或 (p,)
    if pair_scores.ndim == 3:
        pair_scores = pair_scores.squeeze()

    score_matrix = torch.full(
        (1, K, N), -1e7, device=pair_scores.device
    )

    i_idx = pair_indices[:, 0]
    j_idx = pair_indices[:, 1]

    score_matrix[0, i_idx, j_idx] = pair_scores

    return score_matrix

@torch.library.custom_op("clocs::score_aggregator", mutates_args=[])
def score_aggregator(score_matrix: torch.Tensor) -> torch.Tensor:
    """
    对 score_matrix (1, K, N) 在 2D 维度上做最大池化，
    为每个 3D 框得到唯一融合得分。

    Args:
        score_matrix: (1, K, N)
            来自 scatter_to_matrix 的得分矩阵。

    Returns:
        fused_scores: (N)
            每个 3D 框的融合置信度，取 score_matrix 在 K 维的最大值。
    """

    # score_matrix: [1, K, N]
    fused_scores, _ = score_matrix.max(dim=1)  # → [1, N]

    # squeeze → [N]
    fused_scores = fused_scores.squeeze(0)

    return fused_scores
